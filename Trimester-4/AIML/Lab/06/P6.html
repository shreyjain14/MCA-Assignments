def sigmoid(z):
    return 1 / (1 + np.exp(-np.clip(z, -250, 250)))

z = np.linspace(-10, 10, 100)
y = sigmoid(z)
plt.figure(figsize=(8, 6))
plt.plot(z, y, 'b-', linewidth=2)
plt.xlabel('z')
plt.ylabel('sigmoid(z)')
plt.title('Sigmoid Function')
plt.grid(True)
