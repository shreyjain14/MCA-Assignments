evaluation_metrics = {}

for n_clusters in n_clusters_range:
    cluster_labels = results[n_clusters]

    silhouette_avg = silhouette_score(X_scaled, cluster_labels)
    davies_bouldin = davies_bouldin_score(X_scaled, cluster_labels)
    
    evaluation_metrics[n_clusters] = {
        'silhouette_score': silhouette_avg,
        'davies_bouldin_score': davies_bouldin
    }
    
    print(f"n_clusters = {n_clusters}:")
    print(f"  Silhouette Score: {silhouette_avg:.4f}")

fig, axes = plt.subplots(1, 2, figsize=(18, 6))

clusters = list(evaluation_metrics.keys())
silhouette_scores = [evaluation_metrics[k]['silhouette_score'] for k in clusters]
davies_scores = [evaluation_metrics[k]['davies_bouldin_score'] for k in clusters]

axes[0].plot(clusters, silhouette_scores, 'bo-')
axes[0].set_title('Silhouette Score vs Number of Clusters')
axes[0].set_xlabel('Number of Clusters')
axes[0].set_ylabel('Silhouette Score')
axes[0].grid(True, alpha=0.3)
axes[0].set_xticks(clusters)

axes[1].plot(clusters, davies_scores, 'go-')
axes[1].set_title('Davies-Bouldin Score vs Number of Clusters')
axes[1].set_xlabel('Number of Clusters')
axes[1].set_ylabel('Davies-Bouldin Score')
